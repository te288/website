from math import*

def sc(kakudo):
    s = sin(pi*kakudo/180)
    c = cos(pi*kakudo/180)
    return s,c

num = input("開始、終了、間隔を空白区切りで入力＞")

s, g, gap = [int(a) for a in num.split()]

for theta in range(s, g, gap):
    fig = sc(theta) 
    print("sin({0}),cos({0})=".format(theta))
    print(fig)

キーボードから正の整数$n$を入力し，1から$n$までの和と2乗和，3乗和を計算する

def sum_1(number):
    a = 0
    for i in range(1, number+1, 1):
        a += i
    print("1から{}までの和は{}".format(number,a))
    
def sum_sq(number):
    b = 0
    for s in range(1,number+1,1):
        b += s**2
    print("1から{}までの２乗和は{}".format(number,b))
    
def sum_cb(number):
    c = 0
    for t in range(1,number+1,1):
        c += t**3
    print("1から{}までの３乗和は{}".format(number,c))
    
n=int(input("正の整数nを入力＞"))

if n <= 0:
    print("正の整数を入力。")
else:
    sum_1(n)
    sum_sq(n)
    sum_cb(n)

正の整数nを入力＞4
1から4までの和は10
1から4までの２乗和は30
1から4までの３乗和は100

#exp(x)とそのテイラー展開
from math import*
from matplotlib import pyplot as plt


def exp_kinji(n,x):
    kinjiko = 1
    for i in range(1,n+1):
        kinjiko += (x**(i)) / factorial(i)
    c = kinjiko
    return c
        
       
n = int(input("何次の項まで近似？＞"))

x = float(input(" x の値を入力＞"))

ans = exp_kinji(n,x)

print("{0}次まで近似した値は{1}".format(n,ans))

print("Exp({0})={1}".format(x,exp(x)))

練習問題5¶

次のフーリエ級数で表される関数を$0\leq t\leq 6\pi$の範囲で描画し，近似項数$n$による関数の形状を比較する $$ f(n,t)=\frac{4}{\pi}\sum_{k=0}^{n}\frac{1}{2k+1}\sin{\left(2k+1\right)t} $$

In [5]:
from math import*
from matplotlib import pyplot as plt

t=[pi*0.01*x for x in range(0,601)]

def plot_Fourier(n):
    global t
    f_value=[]
    for i in t:
        f_t=0
        for k in range(0,n):
            f_t+=(sin((2*k+1)*i))/(2*k+1)
        f_value.append(4*f_t/pi)
        f_t=0
    plt.plot(t,f_value,label=str(n))
    return

for num in (1, 5, 10, 100, 1000):
    plot_Fourier(num)

plt.title("Fourier series")

plt.grid(True)
plt.xlabel("t")
plt.ylabel("f(x)")
plt.legend(loc='upper right')
plt.show()
