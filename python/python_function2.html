from math import*

def sc(kakudo):
    s = sin(pi*kakudo/180)
    c = cos(pi*kakudo/180)
    return s,c

num = input("")

s, g, gap = [int(a) for a in num.split()]

for theta in range(s, g, gap):
    fig = sc(theta) 
    print("sin({0}),cos({0})=".format(theta))
    print(fig)

def sum_1(number):
    a = 0
    for i in range(1, number+1, 1):
        a += i
    print("1{}{}".format(number,a))
    
def sum_sq(number):
    b = 0
    for s in range(1,number+1,1):
        b += s**2
    print("1{}{}".format(number,b))
    
def sum_cb(number):
    c = 0
    for t in range(1,number+1,1):
        c += t**3
    print("1{}{}".format(number,c))
    
n=int(input("n"))

if n <= 0:
    print("")
else:
    sum_1(n)
    sum_sq(n)
    sum_cb(n)

from math import*
from matplotlib import pyplot as plt

t=[pi*0.01*x for x in range(0,601)]

def plot_Fourier(n):
    global t
    f_value=[]
    for i in t:
        f_t=0
        for k in range(0,n):
            f_t+=(sin((2*k+1)*i))/(2*k+1)
        f_value.append(4*f_t/pi)
        f_t=0
    plt.plot(t,f_value,label=str(n))
    return

for num in (1, 5, 10, 100, 1000):
    plot_Fourier(num)

plt.title("Fourier series")

plt.grid(True)
plt.xlabel("t")
plt.ylabel("f(x)")
plt.legend(loc='upper right')
plt.show()
