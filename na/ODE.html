常微分方程式の時間積分Ⅰ¶

Euler法¶

$y(t)$の微分を次のように定義する。 $$ \frac{dy}{dt} = \lim_{h\to 0}\frac{y(t+h)-y(t)}{h} $$

刻み幅が十分小さいとすれば

$$ \frac{dy}{dt}\approx\frac{y(t+h)-y(t)}{h} $$

1回計算を進めるごとに$\Delta t$だけ時間が進むとして、計算の回数(ステップ数)をnで表すと

$$ \frac{dy}{dt}\approx\frac{y^{n+1}-y^{n}}{\Delta t} $$

$dy/dt = f$とおけば、以下の漸化式をえる。

$$ y^{n+1}=y^{n}+\Delta tf^{n} $$

ここで$f^n$は勾配を表している。例えば$\cos{x}$を解くならば,$f^n=-\sin{x}$

%matplotlib inline

import numpy as np
import math
import matplotlib.pyplot as plt
import japanize_matplotlib

#厳密解をプロットする範囲
t_max = 5*math.pi
t1 = np.arange(0, t_max, 0.01)
y1 = np.cos(t1)

#厳密解のプロット
plt.plot(t1, y1, color = "blue" , label="厳密解")

#漸化式のy^n
def y(x):
    return math.cos(x)

#漸化式のf^n
def f_n(x):
    return -math.sin(x)

#Euler法の詳細(cos(x)の初期値)
dt = 0.5
t2 = 0
y2 = 1

#計算履歴保持用のリスト
t_hist = [0]
y_hist = [1]

while t2 < t_max:
    y2 += f_n(t2)*dt
    t2 += dt
    
    t_hist.append(t2)
    y_hist.append(y2)
    
#Euler法の結果をプロット
plt.plot(t_hist, y_hist, color="red", marker = "+", label="Euler法")



#グラフの表示
plt.title("cos(x)のグラフ-Euler法")
plt.grid(which = 'major')
plt.legend(loc="upper right")
plt.show()

常微分方程式の時間積分Ⅱ¶

Runge-Kutta法¶

勾配を重み付きで平均化し、Euler法より高精度な近似を行う。特に4次精度のものがよく使われている（らしい）。Euler法よりもよい精度で近似できる。$\frac{dy}{dx}=f(x,y)$を近似するには

\begin{align*} &x(t_n+\Delta t) = x(t_n) + \frac{1}{6}\left(k_1+2k_2+2k_3+k_4\right)\\ \text{ただし}\\ &k_1=\Delta t f(x_n, y_n),\quad k_2=\Delta t f(x_n+\frac{\Delta t}{2}, y_n+\frac{k_1}{2})\\ &k_3=\Delta t f(x_n+\frac{\Delta t}{2}, y_n+\frac{k_2}{2})\\ &k_4=\Delta t f(x_n+\frac{\Delta t}{2}, y_n+k_3)\\ \end{align*}

という方法で近似できる。

%matplotlib inline

import numpy as np
import math
import matplotlib.pyplot as plt
import japanize_matplotlib

#厳密解をプロットする範囲
t_max = 5*math.pi
t1 = np.arange(0, t_max, 0.01)
y1 = np.cos(t1)

#厳密解のプロット
plt.plot(t1, y1, color = "blue" , label="厳密解")

#f(x,y)に相当
def df(x):
    return -math.sin(x)

#RK法の詳細(cos(x)の初期値)
dt = 0.5
t2 = 0
y2 = 1

#計算履歴保持用のリスト
t_hist = [0]
y_hist = [1]

while t2 < t_max:
    k1 = dt*df(t2)
    k2 = dt*df(t2 + 0.5*dt)
    k3 = dt*df(t2 + 0.5*dt)
    k4 = dt*df(t2 + dt)
    
    y2 += (k1 +2*k2 + 2*k3 + k4)/6
    t2 += dt
    
    t_hist.append(t2)
    y_hist.append(y2)
    
#Euler法の結果をプロット
plt.plot(t_hist, y_hist, color="red", marker = "+", label="RK法")



#グラフの表示
plt.title("cos(x)のグラフ-RK法")
plt.grid(which = 'major')
plt.legend(loc="upper right")
plt.show()

常微分方程式の時間積分Ⅲ¶

Euler法とRK法の比較¶

速度に比例する空気抵抗(比例定数$k$)を受ける質量$m$の落下速度$v$(初速0)に関する微分方程式 $$ \frac{dv}{dt} = g -\frac{k}{m}v\quad(*) $$ の解析解とEuler法とRK法による数値解を比較してみる。

$(*)$の微分方程式は変数分離で解くことができて、 $$ v = \frac{g}{\mu}\left(1-\mathrm{e}^{-\mu t}\right) $$ となる。($k/m=\mu$)

%matplotlib inline

import numpy as np
import math
import matplotlib.pyplot as plt
import japanize_matplotlib

#各種変数の設定
m = 1
g = 9.81
k = 0.1
mu = k/m

#時間ステップ
dt = 0.5
tmax = 100

#解析解のプロット
t_a = np.arange(0, tmax, 0.01)
#v_a = g*(1 - np.exp(-mu*(t_a)))/mu
v_a = np.sin(t_a)

plt.plot(t_a, v_a, color = "blue", label="解析解")

#dv/dt
def dv(t, v):
    #return g -mu*v
    return math.cos(t)

def v(t, v):
    #return g*(1 - math.exp(-mu*t))/mu
    return math.sin(t)

##Euler法の計算
#Euler法の計算履歴保持
te_hist = [0]
ve_hist = [0]

#ve, teの初期値
ve = 0
te = 0

while te < tmax:
    ve += dt*dv(te, ve)
    te += dt
    
    te_hist.append(te)
    ve_hist.append(ve)

    
#Euler法の計算結果プロット
plt.plot(te_hist, ve_hist, color = "red", label="Euler法")


#RK法の計算
#RK法の計算履歴保持
vr_hist = [0]
tr_hist = [0]

#vr, trの初期値
tr = 0
vr = 0

while tr < tmax:
    k1 = dv(tr, vr)
    k2 = dv(tr + 0.5*dt, vr + 0.5*k1)
    k3 = dv(tr + 0.5*dt, vr + 0.5*k2)
    k4 = dv(tr + dt, vr + k3)
    
    vr += (k1 +2*k2 + 2*k3 + k4)*dt/6
    tr += dt
    
    tr_hist.append(tr)
    vr_hist.append(vr)

#RK法の計算結果プロット
plt.plot(tr_hist, vr_hist, color = "green", label="RK法")


    
    
#結果の表示
#plt.title("質点の落下速度シミュレーション")
plt.title("解析解, Euler法, RK法の比較")
plt.legend()
plt.show()
