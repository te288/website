常微分方程式の時間積分Ⅰ¶

Euler法¶

$y(t)$ $$ \frac{dy}{dt} = \lim_{h\to 0}\frac{y(t+h)-y(t)}{h} $$

$$ \frac{dy}{dt}\approx\frac{y(t+h)-y(t)}{h} $$

1$\Delta t$()n

$$ \frac{dy}{dt}\approx\frac{y^{n+1}-y^{n}}{\Delta t} $$

$dy/dt = f$

$$ y^{n+1}=y^{n}+\Delta tf^{n} $$

$f^n$$\cos{x}$,$f^n=-\sin{x}$

import numpy as np
import math
import matplotlib.pyplot as plt
import japanize_matplotlib

#
t_max = 5*math.pi
t1 = np.arange(0, t_max, 0.01)
y1 = np.cos(t1)

#
plt.plot(t1, y1, color = "blue" , label="")

#y^n
def y(x):
    return math.cos(x)

#f^n
def f_n(x):
    return -math.sin(x)

#Euler(cos(x))
dt = 0.5
t2 = 0
y2 = 1

#
t_hist = [0]
y_hist = [1]

while t2 < t_max:
    y2 += f_n(t2)*dt
    t2 += dt
    
    t_hist.append(t2)
    y_hist.append(y2)
    
#Euler
plt.plot(t_hist, y_hist, color="red", marker = "+", label="Euler")



#
plt.title("cos(x)-Euler")
plt.grid(which = 'major')
plt.legend(loc="upper right")
plt.show()

import numpy as np
import math
import matplotlib.pyplot as plt
import japanize_matplotlib

#
t_max = 5*math.pi
t1 = np.arange(0, t_max, 0.01)
y1 = np.cos(t1)

#
plt.plot(t1, y1, color = "blue" , label="")

#f(x,y)
def df(x):
    return -math.sin(x)

#RK(cos(x))
dt = 0.5
t2 = 0
y2 = 1

#
t_hist = [0]
y_hist = [1]

while t2 < t_max:
    k1 = dt*df(t2)
    k2 = dt*df(t2 + 0.5*dt)
    k3 = dt*df(t2 + 0.5*dt)
    k4 = dt*df(t2 + dt)
    
    y2 += (k1 +2*k2 + 2*k3 + k4)/6
    t2 += dt
    
    t_hist.append(t2)
    y_hist.append(y2)
    
#Euler
plt.plot(t_hist, y_hist, color="red", marker = "+", label="RK")



#
plt.title("cos(x)-RK")
plt.grid(which = 'major')
plt.legend(loc="upper right")
plt.show()

%matplotlib inline

import numpy as np
import math
import matplotlib.pyplot as plt
import japanize_matplotlib

#
m = 1
g = 9.81
k = 0.1
mu = k/m

#
dt = 0.5
tmax = 100

#
t_a = np.arange(0, tmax, 0.01)
#v_a = g*(1 - np.exp(-mu*(t_a)))/mu
v_a = np.sin(t_a)

plt.plot(t_a, v_a, color = "blue", label="")

#dv/dt
def dv(t, v):
    #return g -mu*v
    return math.cos(t)

def v(t, v):
    #return g*(1 - math.exp(-mu*t))/mu
    return math.sin(t)

##Euler
#Euler
te_hist = [0]
ve_hist = [0]

#ve, te
ve = 0
te = 0

while te < tmax:
    ve += dt*dv(te, ve)
    te += dt
    
    te_hist.append(te)
    ve_hist.append(ve)

    
#Euler
plt.plot(te_hist, ve_hist, color = "red", label="Euler")


#RK
#RK
vr_hist = [0]
tr_hist = [0]

#vr, tr
tr = 0
vr = 0

while tr < tmax:
    k1 = dv(tr, vr)
    k2 = dv(tr + 0.5*dt, vr + 0.5*k1)
    k3 = dv(tr + 0.5*dt, vr + 0.5*k2)
    k4 = dv(tr + dt, vr + k3)
    
    vr += (k1 +2*k2 + 2*k3 + k4)*dt/6
    tr += dt
    
    tr_hist.append(tr)
    vr_hist.append(vr)

#RK
plt.plot(tr_hist, vr_hist, color = "green", label="RK")


    
    
#
#plt.title("")
plt.title(", Euler, RK")
plt.legend()
plt.show()
